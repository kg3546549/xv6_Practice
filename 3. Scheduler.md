# 1. SSU 스케쥴러 구현 내용

1. 스케쥴러를 2번 순환하도록 구현
2. pTable에서 선택된 Process의 PID를 i로 나누어 PID를 홀수/짝수로 판단함.
3. i가 0일 떄는 PID가 짝수인 프로세스를 Continue하고, i가 1일 때는 PID가 홀수인 프로세스를 Continue함

```c
if( p->pid % 2 == i) //If PID is Even
          continue;
```

## 전체 코드

### 스케쥴러 코드

```c
#define SSUScheduler 0
#define DEBUG 0

void
scheduler(void)
{
  struct proc *p;
  struct cpu *c = mycpu();
  c->proc = 0;
  #if SSUScheduler
  int i=0, cnt=0;
  cprintf("[SSU Scheduler is enabled]\n");
  #else
  cprintf("[SSU Scheduler is disabled]\n");
  #endif
  
  for(;;){
    // Enable interrupts on this processor.
    sti();

    // Loop over process table looking for process to run.
    acquire(&ptable.lock);
    #if SSUScheduler
    for(i=0;i<2;i++) {
      cnt = 0;
    #endif
      
      for(p = ptable.proc; p < &ptable.proc[NPROC]; p++){
        if(p->state != RUNNABLE)
          continue;
        
        #if SSUScheduler
        //GEON: 짝수 프로세스와 홀수 프로세스 번갈아 스케줄링
        if( p->pid % 2 == i) //If PID is Even or Odd
          continue;
        #endif

        //GEON: runnable_start가 0이 아니면, runnable_start에서 현재 틱까지의 시간 차이를 waiting_time에 추가
        p->waiting_time += (ticks - p->runnable_start);
        p->runnable_start = 0;

        //GEON: 프로세스가 처음 실행되는 경우, 시작 시간을 기록
        if(p->has_run == 0) {
          p->response_time = ticks;
          p->has_run = 1; //GEON: has_run을 1로 설정
        }

        // Switch to chosen process.  It is the process's job
        // to release ptable.lock and then reacquire it
        // before jumping back to us.
        c->proc = p;
        switchuvm(p); //GEON: 유저 프로세스의 페이지 테이블로 전환
        p->state = RUNNING;

        #if SSUScheduler
        cnt++;
        #endif

        #if DEBUG
        cprintf("pid : %d(%s) have been just schedled\n", p->pid, p->name);
        #endif

        swtch(&(c->scheduler), p->context);
        switchkvm(); //GEON: 커널 프로세스의 페이지 테이블로 전환
        
        // Process is done running for now.
        // It should have changed its p->state before coming back.
        c->proc = 0;
      }
      
      #if DEBUG    
      if(cnt > 0) cprintf("\n============================\n");
      #endif
    #if SSUScheduler
    }
    #endif
    release(&ptable.lock);

  }
}
```

```c
#define PROC_NUM 50
void scheduler_func(void)
{
    int mainProcPid = getpid();
    int pid, tick0, tick1;
    printf(1, "start scheduler_test - PID : %d \n", mainProcPid);

    for (int i = 0; i < PROC_NUM; i++)
    {
        pid = fork();
        if(pid != 0) {
            printf(1, "Fork Process[%d]\n", pid);
        }
        
        if (pid == 0) //Child Process
        {
            tick0 = uptime();
            tick1 = uptime();
            while (tick1 - tick0 < 100) //실행하고 100 tick 후 종료
                tick1 = uptime();

            exit();
        }
    }
    for (int i = 0; i < PROC_NUM; i++)
    {
        wait();
    }
    printf(1, "end scheduler_test\n");
}
int main(void)
{
    scheduler_func();
    
    exit();
}
```

# 2. 성능 분석

<aside>
💡

**대기 시간 (Waiting Time)**
  - 프로세스가 `RUNNABLE` 상태에서 실제 CPU를 얻기까지 기다린 총 시간

**턴어라운드 시간(Turnaround Time)**
  - 프로세스가 생성된 순간부터 종료될 때까지의 전체 시간

**응답 시간 (Response Time)**
  - 프로세스가 생성되고 나서 처음 `RUNNING` 상태가 될 때까지 걸린 시간

</aside>

| 평균(tick) | Wait Time | Turn Around Time | Response Time |
| --- | --- | --- | --- |
| SSU Scheduler | 370.44 | 125.98 | 186.7 |
| xv6 Scheduler | 336.56 | 126.8 | 233.88 |

1 tick **≈** 10ms

## 구현 내용

### proc 구조체에 시간 변수 추가

```c
struct proc {
  uint sz;                     // Size of process memory (bytes)
  pde_t* pgdir;                // Page table
  char *kstack;                // Bottom of kernel stack for this process
  enum procstate state;        // Process state
  int pid;                     // Process ID
	//...생략...
	
  //GEON: Wating Time Analyze
  int runnable_start;
  int waiting_time;

  //GEON: Turnaround Time Analyze
  int start_time;
  int end_time;

  //GEON: Response Time Analyze
  int response_time;
  int has_run;
};
```

### 대기 시간 분석(Waiting Time)

<aside>
💡

**대기 시간 (Waiting Time)**
  - 프로세스가 `RUNNABLE` 상태에서 실제 CPU를 얻기까지 기다린 총 시간

</aside>

1. 프로세스가 실행 중 스케쥴링되어 sched에서 스케쥴러로 Context Switching이 발생할 때 기록 시작.

```c
void
sched(void)
{
	//...생략...
	//GEON: 대기 시작 시간 기록
	if (p->state == RUNNABLE && p->runnable_start == 0)
		p->runnable_start = ticks;
	//...생략...
}
```

1. 스케쥴러가 RUNNABLE한 프로세스를 발견하였을 때 대기시간 저장.

```c
void
scheduler(void)
{
	//...생략...
	//GEON: 스케쥴러가 RUNNABLE한 프로세스를 발견하였을 때까지로 대기시간 종료.
	p->waiting_time += (ticks - p->runnable_start);
	p->runnable_start = 0;
	//...생략...
}
```

### 턴어라운드 시간 분석(TurnAround Time)

<aside>
💡

**턴어라운드 시간(Turnaround Time)**
  - 프로세스가 생성된 순간부터 종료될 때까지의 전체 시간

</aside>

1. 프로세스가 fork()되면 시작 시간을 기록

```c
int
fork(void)
{
	//...생략...
	np->start_time = ticks; //GEON: 시작 시간 기록
	//...생략...
}
```

1. 프로세스가 exit()되면 종료 시간을 기록

```c
int
exit(void)
{
	//...생략...
	curproc->end_time = ticks;
	//...생략...
}
```

### 응답 시간 분석(Response Time)

<aside>
💡

**응답 시간 (Response Time)**
  - 프로세스가 생성되고 나서 처음 `RUNNING` 상태가 될 때까지 걸린 시간

</aside>

1. 프로세스가 fork()되면 시작 시간을 기록

```c
int
fork(void)
{
	//...생략...
	//GEON: 응답시간 분석을 위해 초기화
  np->has_run = 0; 
  np->response_time = 0;
	//...생략...
}  
```

1. 스케쥴러에 의해 최초로 RUNNING 상태가 되면 응답 시간을 기록

```c
void
scheduler(void)
{
	//...생략...
	//GEON: 프로세스가 처음 실행되는 경우, 시작 시간을 기록
  if(p->has_run == 0) {
    p->response_time = ticks;
    p->has_run = 1; //GEON: has_run을 1로 설정
  }
  //...생략...
}
```

## **xv6 실행 및 초기화(main.c) 분석**

### 최초 실행 : main.c - int main(void)

```c
int
main(void)
{
  kinit1(end, P2V(4*1024*1024)); // phys page allocator
  kvmalloc();      // kernel page table
  mpinit();        // detect other processors
  lapicinit();     // interrupt controller
  seginit();       // segment descriptors
  picinit();       // disable pic
  ioapicinit();    // another interrupt controller
  consoleinit();   // console hardware
  uartinit();      // serial port
  pinit();         // process table
  tvinit();        // trap vectors
  binit();         // buffer cache
  fileinit();      // file table
  ideinit();       // disk 
  startothers();   // start other processors
  kinit2(P2V(4*1024*1024), P2V(PHYSTOP)); // must come after startothers()
  userinit();      // first user process
  mpmain();        // finish this processor's setup
}
```

xv6 OS가 부팅하면서 초기화 작업을 진행합니다. 프로세스 초기화 순서는 위 코드와 같습니다.

프로세스 할당 관련한 초기화 과정은 아래 함수에서 실행됩니다.

`pinit()` - 프로세스 테이블 관련 초기화
`userinit()` - initcode 프로세스를 pTable에 추가함.
`mpmain()` - 스케쥴러를 실행

스케쥴러가 실행되면, `userinit()` 에서 pTable에 추가한 initcode 프로세스를 실행합니다.
initcode 프로세스는 initcode.S 에 선언되어 있으며, 이는 **init 프로세스를 실행합니다.**

## 스케쥴러 함수 내부 구현 분석

스케쥴러 함수가 호출되면, 

1. 스케쥴러가 pTable을 순환하며 `for(p = ptable.proc; p < &ptable.proc[NPROC]; p++){`
Runnable 한 프로세스를 확인함. `if(p->state != RUNNABLE)`
2. 해당 프로세스와 스케쥴러를 Context Switching 함 `swtch(&(c->scheduler), p->context);`
3. 프로세스가 실행 도중 **타이머 인터럽트**가 걸려 Scheduler로 Context Switching 됨
(**스케줄링 호출 순서 분석** 챕터 참고)
4. 다시 다른 프로세스를 찾기 위해 1번으로 돌아감.

## 스케쥴러 호출 순서 분석

### Timer Interrupt 초기화

부팅 당시 **`lapicinit()`**에서 인터럽트 관련 설정을 초기화합니다.

**lapic**: local APIC(Advanced **Programmable Interrupt Controller**)

```c
#define TICR    (0x0380/4)   // Timer Initial Count
#define TCCR    (0x0390/4)   // Timer Current Count
#define TDCR    (0x03E0/4)   // Timer Divide Configuration

lapicw(TDCR, X1);
lapicw(TIMER, PERIODIC | (T_IRQ0 + IRQ_TIMER));
lapicw(TICR, 10000000);
```

`lapicw(TDCR, X1);` 
**TCDR**: Timer Divide Configuration Registor
클럭 분주비를 1로 설정

`lapicw(TIMER, PERIODIC | (T_IRQ0 + IRQ_TIMER));` 
타이머 인터럽트 번호 설정 

`lapicw(TICR, 10000000);` 
**TICR**: Timer Initial Count Register
오실레이터에서 발생하는 클럭이 10,000,000회 카운트 되어야 인터럽트 발생.

### Timer Interrupt 발생

- 발생한 인터럽트는 `trap()` 함수로 전달
- `T_IRQ0 + IRQ_TIMER`인 경우(타이머 인터럽트인 경우) → `yield()` 호출
- 현재 프로세스는 CPU 양보 → 스케줄러와 Context Switching(`sched()`)

함수 호출 순서 : **`trap()`** → **`yield()`**  → **`sched()`**

## Shell에서 application을 실행하는 순서 분석

### pTable 구조

xv6가 실행되면 64(`NPROC`)개까지 설정 가능한 프로세스 테이블인 `ptable` 이 다음과 같이 선언됩니다.

```c
#define NPROC        64  // maximum number of processes

struct {
  struct spinlock lock; //pinit()에서 초기화 한 프로세스 테이블 락, 임계영역 보호를 위함.
  struct proc proc[NPROC]; //실제 프로세스 테이블 배열
} ptable;
```

### fork() 함수 동작

User가 fork()를 실행하면 할당된 pid를 반환합니다.

내부적으로 아래와 같은 순서로 구현되어있습니다.

1. fork를 호출하면 내부에 **allocproc()** 함수에서 pTable에 EMBRYO state인 프로세스를 할당한다.
2. 이후, 해당 프로세스를 부모 프로세스와 동일한 메모리 영역을 가리키게 한 다음,
3. state를 **RUNNABLE**로 설정하여 프로세스 초기화를 완료한다.

**fork함수의 내부**

```c
int
fork(void)
{
  int i, pid;
  struct proc *np;//새로운 프로세스 구조체 (newProcess)
  struct proc *curproc = myproc(); //현재 프로세스(부모 프로세스)

  // Allocate process.
  if((np = allocproc()) == 0){ //new Process에 새로운 프로세스를 할당함.
    return -1;
  }
...
```

위 코드를 보면 `allocproc()` 을 통해 새로운 프로세스를 할당합니다.

**allocproc() 함수 내부**

allocate함수에서는 위에서 말한 것과 같이 **pTable**에서 사용하지 않는 영역을 찾아 해당 영역을 
**EMBRYO(태아)** 상태로 설정하고, 해당 프로세스에 커널 스택을 할당합니다**(kalloc)**.

```c
static struct proc*
allocproc(void)
{
  struct proc *p;
  char *sp;

  acquire(&ptable.lock); //임계영역 보호를 위한 Lock 설정

	//pTable을 순회하며 table에 사용되지 않는 영역이 있는지 찾아서 found 프로시저를 실행한다.
  for(p = ptable.proc; p < &ptable.proc[NPROC]; p++) 
    if(p->state == UNUSED)
      goto found;

  release(&ptable.lock);
  return 0;

found:
  
  p->state = EMBRYO; //EMBRYO(태아) 상태로 State 변환
  p->pid = nextpid++; //PID 부여

  release(&ptable.lock); //pTable Lock 해제

  // Allocate kernel stack.
  // 해당 프로세스의 커널 스택을 할당한다.
  if((p->kstack = kalloc()) == 0){
    p->state = UNUSED;
    return 0;
  }
  sp = p->kstack + KSTACKSIZE;

  //... 이하 초기화 코드 생략 ...
  return p; //프로세스 반환
}
```

allocproc()이 실행되면, ptable의 proc 배열을 순회한다.

proc배열은 위에서 설명했다시피 실제 현재 할당된 프로세스들이 있는 테이블이다.

순회 중 **사용중이지 않은(Unused = 사용 가능한)** ptable이 있다면 **found** 프로시저로 이동한다.

found 프로시저에서는 찾은 프로세스 테이블을 EMBRYO(태아)상태로 변환하고, 
해당 테이블에 pid를 할당한다.(nextpid로 설정하고 카운터를 올린다.)

또한, 해당 테이블에 새로 생성한 프로세스에 커널 영역을 할당하고, 초기화한다.

따라서, pTable에는 EMBRYO로 설정된 프로세스가 존재하게 된다.

이로써 fork를 이용한 프로세스 할당이 완료된다.

추후에 스케쥴러에서 pTable을 실행하며 실행할 수 있는 프로세스를 확인한다.

```c
for(p = ptable.proc; p < &ptable.proc[NPROC]; p++){
        if(p->state != RUNNABLE) //fork하여 RUNNABLE한 프로세스가 있는 지 확인함.
          continue;
```

### $ (Shell)창에서 입력한 명령어가 실행되는 방식

부팅 시에 모든 초기화 작업이 완료되면, init() 프로세스는 sh(shell)프로세스를 실행한다.

```c
int
main(void)
{
  int pid, wpid;

  if(open("console", O_RDWR) < 0){
    mknod("console", 1, 1);
    open("console", O_RDWR);
  }
  dup(0);  // stdout
  dup(0);  // stderr

  for(;;){
    printf(1, "init: starting sh\n");
    pid = fork(); //fork하여 자식 프로세스를 만든다.
    //fork를 한 후 부모 프로세스는 아래 while문으로 간다.
    if(pid < 0){ //pid가 
      printf(1, "init: fork failed\n");
      exit();
    }
    //내가 자식프로세스라면. (fork를 실행하여 생성되는 자식프로세스의 pid는 0이다)
    if(pid == 0){
      exec("sh", argv); //sh 프로세스를 실행한다.
      //exec에 성공하면 아래 코드는 실행하지 않는다.
      
      printf(1, "init: exec sh failed\n"); //따라서 exec()에 실패하면 여기를 실행한다.
      exit(); //
    }
    while((wpid=wait()) >= 0 && wpid != pid) //부모 프로세스는 wait 상태가 된다.
      printf(1, "zombie!\n");
  }
}
```

fork를 하면 나와 같은 프로세스를 자식 프로세스로 만든다.

자식 프로세스는 본인의 pid를 모르기 때문에, pid가 0이다. 
(pid를 알기 위해선 **`getpid()`** System Call이 필요하다)

pid가 0일 때 (내가 자식 프로세스 일 때) **`exec(”sh”)`** 하여 sh 프로세스를 실행한다.

```c
int
getcmd(char *buf, int nbuf)
{
  printf(2, "$ ");
  memset(buf, 0, nbuf);
  //printf(2, "gets wait\n");

  gets(buf, nbuf);

  //printf(2, "gets end\n");
  if(buf[0] == 0) // EOF
    return -1;
  return 0;
}

int
main(void)
{
  static char buf[100];
  int fd;

  // Ensure that three file descriptors are open.
  while((fd = open("console", O_RDWR)) >= 0){
    if(fd >= 3){
      close(fd);
      break;
    }
  }

  // Read and run input commands.
  while(getcmd(buf, sizeof(buf)) >= 0){
    if(buf[0] == 'c' && buf[1] == 'd' && buf[2] == ' '){
      // Chdir must be called by the parent, not the child.
      buf[strlen(buf)-1] = 0;  // chop \n
      if(chdir(buf+3) < 0)
        printf(2, "cannot cd %s\n", buf+3);
      continue;
    }
    if(fork1() == 0)
      runcmd(parsecmd(buf));
    wait();
  }
  exit();
}
```

`getcmd()` 에서 “$”를 출력 후 `gets()` 를 이용하여 키보드 입력을 받는다.

입력받은 명령어를 `parsecmd()`를 통해 파싱하여 `runcmd()`에서 실행한다.
